    return _.map(a, function(b){
        return b['Course']
    })
})

var levelCount = _.mapValues(courseLevels, function(level){
    count = 0
    return _.map(level, function(n){
    	if(n >= 1000 && n < 2000){
    		count++
		}
		return count
    })
})

	return _.max(n)
})
